# Range Chart Analyzer runs on the Python standard library alone.
# The three optional deps below improve the experience but are not required:
#
# Pillow: enables the GUI image thumbnail preview and client-side downscaling
#          of very large images.
# sv_ttk: swaps the Tkinter "clam" theme for a Windows 11 Fluent-style
#          flat/modern skin (buttons, inputs, combobox, notebook tabs).
#          Without it the app falls back to ttk's default clam theme.
# pywebview: enables `python main.py --ui modern`, a native-window wrapper
#          around the existing web frontend. On Linux you also need the
#          system package `webkit2gtk-4.0` (e.g. `apt install
#          libwebkit2gtk-4.0-dev`). Without pywebview the modern mode
#          falls back to opening the local URL in the default browser.
Pillow>=9.0 ; python_version >= "3.8"
sv_ttk>=2.5  ; python_version >= "3.8"
pywebview>=5.0 ; python_version >= "3.8"
# PySide6 + qfluentwidgets: enable the modern Fluent desktop GUI
#   (python main.py --ui fluent). Large (~150MB); fully optional — the
#   stdlib Tkinter GUI (python main.py) remains the default and the
#   Fluent launcher falls back to it automatically if these are missing.
PySide6>=6.6 ; python_version >= "3.9"
PySide6-Fluent-Widgets>=1.5 ; python_version >= "3.9"
# PySide6-Addons — pulls in QtWebEngineWidgets for the History detail
# dialog (gui_fluent_history_detail.HistoryDetailDialog). Heavy
# (~+50MB on top of PySide6) but optional: the rest of the Fluent GUI
# works without it; double-clicking a history row will show a no-engine
# fallback message instead of the rich result render.
PySide6-Addons>=6.6 ; python_version >= "3.9"
# XLSX export — enables the "Export Excel" button in the Fluent GUI and
#   the `rca_core.exporter.to_xlsx` function. Optional: the rest of the
#   app (CSV / JSON / TSV) works fine without it.
openpyxl>=3.1 ; python_version >= "3.8"
# Token-usage charts — rendered on the Usage page. Falls back to plain
#   text tables if missing.
matplotlib>=3.5 ; python_version >= "3.8"
